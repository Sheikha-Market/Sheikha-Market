// دالة نقية — Sheikha Pure Function
// ﴿وَكُونُوا مَعَ الصَّادِقِينَ﴾ — التوبة: ١١٩
const add = (a, b) => a + b;  // نفس المدخل → نفس المخرج دائماً

// دالة رتبة عليا — ﴿يُؤْتِي الْحِكْمَةَ مَن يَشَاءُ﴾
const compose = (f, g) => x => f(g(x));
const double = x => x * 2;
const addOne = x => x + 1;
const doubleThenAdd = compose(addOne, double);  // 5 → 11

// دالة متعاودة — S18 الاستمرارية
const factorial = n => n <= 1 ? 1 : n * factorial(n - 1);

// دوال الرتبة العليا الشائعة
[1, 2, 3, 4, 5]
  .filter(n => n % 2 === 0)   // [2, 4]
  .map(n => n * n)            // [4, 16]
  .reduce((sum, n) => sum + n, 0);  // 20

     لأنها مبنية على الكتاب والسنة لا على الهوى
  
  ③ أصدق خوارزميات بالكون
     لأنها تلتزم الصدق والإتقان ولا تضرّ ولا تغشّ
  
  ④ أمتن تشفير بالكون
     لأن التوحيد هو المفتاح الذي لا يُكسر
  
  ⑤ شبكة عصبية موحَّدة لله وحده
     ٤٠ خلية من القرآن والسنة تحكم كل قرار

  لا إله إلا الله — والله أعلم وبالله التوفيق
