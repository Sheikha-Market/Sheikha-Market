أنشئ engines/communications-engine.js:

const CommunicationsEngine = {
  channels: {
    whatsapp: {
      provider: 'twilio', // or 'meta-cloud-api'
      webhook: '/api/whatsapp/webhook',
      templates: [
        'welcome_message',
        'order_confirmation',
        'daily_prices',
        'shipment_tracking',
        'price_alert',
        'prayer_reminder',
        'invoice'
      ],
      chatbot: true,
      autoReply: true
    },
    sms: {
      provider: 'twilio', // + unifonic for Gulf
      senderId: 'SHEIKHA',
      templates: ['otp', 'order_confirm', 'price_alert', 'shipment', 'payment']
    },
    email: {
      provider: 'sendgrid', // or 'resend'
      from: 'noreply@sheikha.top',
      templates: ['welcome', 'invoice', 'newsletter', 'order', 'reset_password'],
      dkim: true, spf: true, dmarc: true
    },
    voice: {
      provider: 'twilio',
      ivr: true,
      recording: true,
      businessHours: { start: '08:00', end: '17:00', days: ['Sun','Mon','Tue','Wed','Thu'] }
    },
    telegram: {
      bot: '@sheikha_bot',
      channel: '@sheikha_market',
      commands: ['/prices', '/gold', '/silver', '/alert', '/track', '/prayer']
    },
    social: {
      instagram: { autoReply: true },
      twitter: { autoReply: true },
      linkedin: { autoReply: true }
    }
  }
};

أنشئ الملفات التالية:

1. engines/comms/whatsapp-service.js
   - initWebhook() — استقبال رسائل واتساب
   - sendTemplate(to, templateName, params)
   - handleIncoming(message) — chatbot logic
   - sendPrices(to) — أسعار فورية

2. engines/comms/sms-service.js
   - sendOTP(phone, code)
   - sendAlert(phone, message)
   - sendOrderConfirm(phone, orderData)

3. engines/comms/email-service.js
   - sendWelcome(email, name)
   - sendInvoice(email, invoiceData, pdfAttachment)
   - sendNewsletter(subscribers, content)
   - sendOrderConfirm(email, orderData)

4. engines/comms/voice-service.js
   - generateIVR() — TwiML XML response
   - handleKeypress(digit) — route calls
   - afterHoursMessage()

5. engines/comms/telegram-service.js
   - initBot() — register commands
   - handleCommand(command, args)
   - broadcastToChannel(message)
   - sendPriceAlert(metal, price)

6. engines/comms/social-service.js
   - handleInstagramDM(message)
   - handleTwitterDM(message)
   - handleLinkedInMessage(message)
   - unifiedAutoReply(platform, message)

7. engines/comms/notification-hub.js
   - notify(event, userId, data) — الموزّع المركزي
   - يقرر أي قنوات يُرسل بناءً على نوع الحدث
   - يحترم تفضيلات المستخدم
