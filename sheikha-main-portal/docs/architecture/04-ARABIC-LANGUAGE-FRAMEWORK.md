### 1.2 علوم اللغة العربية

| العلم | التعريف | الفائدة في المنظومة |
|-------|---------|---------------------|
| **النحو** | علم إعراب الكلام وتركيب الجمل | فهم وتحليل النصوص |
| **الصرف** | علم بنية الكلمة وتحولاتها | تحليل الكلمات واشتقاقها |

| **المنطق** | علم الاستدلال والقياس | التفكير السليم والتحليل |

## 4. علم البلاغة {#علم-البلاغة}

### 4.1 علم البيان

```
┌──────────────────────────────────────────────────────────────────────────────────┐
│                              علم البيان                                          │
├──────────────────────────────────────────────────────────────────────────────────┤
│                                                                                  │
│  ┌─────────────────────────────────────────────────────────────────────────┐   │
│  │                              التشبيه                                     │   │
│  │  ═══════════════════════════════════════════════════════════════════   │   │
│  │                                                                          │   │
│  │  الأركان: المشبه + المشبه به + أداة التشبيه + وجه الشبه                 │   │
│  │                                                                          │   │
│  │  ┌────────────┬─────────────────────────────────────────────────────┐  │   │
│  │  │ النوع      │ المثال                                              │  │   │
│  │  ├────────────┼─────────────────────────────────────────────────────┤  │   │
│  │  │ مرسل مفصل │ العلم كالنور في الهداية                             │  │   │
│  │  │ مرسل مجمل │ أنت كالشمس                                          │  │   │
│  │  │ مؤكد مفصل │ العلم نور يهدي                                      │  │   │
│  │  │ بليغ      │ أنت شمس (حُذفت الأداة ووجه الشبه)                   │  │   │
│  │  │ تمثيلي    │ مَثَلُ الَّذِينَ يُنفِقُونَ... كَمَثَلِ حَبَّةٍ...      │  │   │
│  │  └────────────┴─────────────────────────────────────────────────────┘  │   │
│  └─────────────────────────────────────────────────────────────────────────┘   │
│                                                                                  │
│  ┌─────────────────────────────────────────────────────────────────────────┐   │
│  │                            الاستعارة                                     │   │
│  │  ═══════════════════════════════════════════════════════════════════   │   │
│  │                                                                          │   │
│  │  • تصريحية: صُرح فيها بالمشبه به                                        │   │
```

### 5.2 أمثلة برمجية بالعربية

```javascript
// ════════════════════════════════════════════════════════════════════════
//  مثال 1: تعريف متغيرات
// ════════════════════════════════════════════════════════════════════════

متغير الاسم = "أحمد";
ثابت العمر = 25;
متغير المهنة = "مهندس";

// ════════════════════════════════════════════════════════════════════════
//  مثال 2: دالة بسيطة
// ════════════════════════════════════════════════════════════════════════

دالة احسب_المجموع(أ، ب) {
    أرجع أ + ب;
}

ثابت النتيجة = احسب_المجموع(5، 3);
اطبع(النتيجة); // 8

// ════════════════════════════════════════════════════════════════════════
//  مثال 3: جملة شرطية
// ════════════════════════════════════════════════════════════════════════

إذا (العمر > 18) {
    اطبع("بالغ");
} وإلا_إذا (العمر > 12) {
    اطبع("مراهق");
} وإلا {
    اطبع("طفل");
}

// ════════════════════════════════════════════════════════════════════════
//  مثال 4: حلقة تكرارية
// ════════════════════════════════════════════════════════════════════════

لكل (متغير ع = 1؛ ع <= 10؛ ع++) {
    اطبع(ع);
}

// أو باستخدام طالما
متغير عداد = 0;
طالما (عداد < 5) {
    اطبع(عداد);
    عداد++;
}

// ════════════════════════════════════════════════════════════════════════
//  مثال 5: صنف (كلاس)
// ════════════════════════════════════════════════════════════════════════

صنف شخص {
    منشئ(الاسم، العمر) {
        هذا.الاسم = الاسم;
        هذا.العمر = العمر;
    }
    
    قدم_نفسك() {
        اطبع("أنا " + هذا.الاسم + " وعمري " + هذا.العمر);
    }
    
    هل_بالغ() {
        أرجع هذا.العمر >= 18;
    }
}

ثابت شخص_جديد = جديد شخص("محمد"، 30);
شخص_جديد.قدم_نفسك();

// ════════════════════════════════════════════════════════════════════════
//  مثال 6: التعامل مع المصفوفات
// ════════════════════════════════════════════════════════════════════════

ثابت الأسماء = ["أحمد"، "محمد"، "علي"، "فاطمة"];

// إضافة عنصر
الأسماء.أضف("خالد");

// التكرار على المصفوفة
الأسماء.لكل_عنصر((اسم) => {
    اطبع("مرحباً " + اسم);
});

// الترشيح
ثابت الأسماء_الطويلة = الأسماء.رشح((اسم) => اسم.طول > 3);

// ════════════════════════════════════════════════════════════════════════
//  مثال 7: معالجة الأخطاء
// ════════════════════════════════════════════════════════════════════════

حاول {
    // كود قد يسبب خطأ
    ثابت نتيجة = اقسم(10، 0);
} التقط (خطأ) {
    اطبع("حدث خطأ: " + خطأ.رسالة);
} أخيراً {
    اطبع("تم الانتهاء");
}
```

## 7. التكامل مع المنظومة {#التكامل}

### 7.1 استخدام محرك اللغة العربية

```javascript
// استيراد المحرك
const { محرك_العربية } = require('./lib/arabic-language-engine');

// تحليل جملة
const تحليل = محرك_العربية.حلل_الجملة("كتبَ الطالبُ الدرسَ");
console.log(تحليل);

// تحليل كلمة
const تحليل_كلمة = محرك_العربية.حلل_الكلمة("يكتبون");
console.log(تحليل_كلمة);

// ترجمة كود عربي إلى جافاسكربت
const كود_عربي = `
دالة مرحبا(الاسم) {
    اطبع("مرحباً " + الاسم);
}
`;
const كود_إنجليزي = محرك_العربية.ترجم_إلى_جافاسكربت(كود_عربي);
console.log(كود_إنجليزي);

// شرح قاعدة نحوية
const شرح = محرك_العربية.اشرح_قاعدة("الفاعل");
console.log(شرح);

// قياس منطقي
const قياس = محرك_العربية.قياس_منطقي(
    "كل إنسان فانٍ",
    "سقراط إنسان"
);
console.log(قياس);
```
