### 1.2 الأصول الشرعية المعتمدة

| المصدر | الوصف | التطبيق في المنظومة |
|--------|-------|---------------------|
| **القرآن الكريم** | المصدر الأول للتشريع | آيات البيع والتجارة والعقود |
| **السنة النبوية** | شرح وتفصيل القرآن | أحاديث المعاملات والأخلاق |
| **الإجماع** | اتفاق العلماء | القواعد المتفق عليها |
| **القياس** | الاجتهاد الشرعي | المستجدات التقنية |
| **المصالح المرسلة** | تحقيق المصلحة | الابتكار في الخدمات |

```javascript
// lib/sharia-compliance.js

/**
 * ═══════════════════════════════════════════════════════════════════════════════
 *  محرك التحقق الشرعي - Sharia Compliance Engine
 * ═══════════════════════════════════════════════════════════════════════════════
 */

const SHARIA_RULES = {
    // قواعد الربا
    riba: {
        id: 'RIBA_001',
        name: 'تحريم الربا',
        severity: 'CRITICAL',
        evidence: 'وَأَحَلَّ اللَّهُ الْبَيْعَ وَحَرَّمَ الرِّبَا',
        checks: [
            { type: 'NO_INTEREST', description: 'لا فائدة على التأخير' },
            { type: 'NO_RIBA_FADL', description: 'لا ربا الفضل في المعادن' },
            { type: 'NO_RIBA_NASIAH', description: 'لا ربا النسيئة' }
        ]
    },
    
    // قواعد الغرر
    gharar: {
        id: 'GHARAR_001',
        name: 'منع الغرر الفاحش',
        severity: 'HIGH',
        evidence: 'نهى النبي ﷺ عن بيع الغرر',
        checks: [
            { type: 'CLEAR_DESCRIPTION', description: 'وصف واضح للمنتج' },
            { type: 'KNOWN_PRICE', description: 'سعر معلوم' },
            { type: 'DELIVERABLE', description: 'قابل للتسليم' }
        ]
    },
    
    // قواعد الغش
    ghish: {
        id: 'GHISH_001',
        name: 'تحريم الغش',
        severity: 'CRITICAL',
        evidence: 'من غشنا فليس منا',
        checks: [
            { type: 'TRUE_SPECS', description: 'مواصفات حقيقية' },
            { type: 'NO_DEFECT_HIDING', description: 'لا إخفاء للعيوب' },
            { type: 'ACCURATE_WEIGHT', description: 'وزن دقيق' }
        ]
    },
    
    // قواعد الاحتكار
    ihtikar: {
        id: 'IHTIKAR_001',
        name: 'منع الاحتكار',
        severity: 'HIGH',
        evidence: 'لا يحتكر إلا خاطئ',
        checks: [
            { type: 'NO_HOARDING', description: 'لا تخزين للمضاربة' },
            { type: 'FAIR_SUPPLY', description: 'توزيع عادل' }
        ]
    }
};

class ShariaComplianceEngine {
    constructor() {
        this.rules = SHARIA_RULES;
        this.auditLog = [];
    }
    
    /**
     * التحقق الشامل من المعاملة
     */
    async validateTransaction(transaction) {
        const result = {
            transactionId: transaction.id,
            timestamp: new Date().toISOString(),
            status: 'PENDING',
            checks: [],
            overallScore: 100,
            fatwa: null
        };
        
        // 1. فحص الربا
        const ribaCheck = this.checkRiba(transaction);
        result.checks.push(ribaCheck);
        if (!ribaCheck.passed) {
            result.status = 'REJECTED';
            result.overallScore = 0;
            return this.finalizeResult(result);
        }
        
        // 2. فحص الغرر
        const ghararCheck = this.checkGharar(transaction);
        result.checks.push(ghararCheck);
        if (!ghararCheck.passed && ghararCheck.severity === 'CRITICAL') {
            result.status = 'REJECTED';
            result.overallScore = 0;
            return this.finalizeResult(result);
        }
        
        // 3. فحص الغش
        const ghishCheck = this.checkGhish(transaction);
        result.checks.push(ghishCheck);
        if (!ghishCheck.passed) {
            result.status = 'REJECTED';
            result.overallScore = 0;
            return this.finalizeResult(result);
        }
        
        // 4. فحص الاحتكار
        const ihtikarCheck = this.checkIhtikar(transaction);
        result.checks.push(ihtikarCheck);
        
        // حساب النتيجة النهائية
        result.overallScore = this.calculateScore(result.checks);
        result.status = result.overallScore >= 80 ? 'APPROVED' : 
                       result.overallScore >= 50 ? 'NEEDS_REVIEW' : 'REJECTED';
        
        return this.finalizeResult(result);
    }
    
    /**
     * فحص الربا
     */
    checkRiba(transaction) {
        const check = {
            rule: this.rules.riba,
            passed: true,
            issues: [],
            suggestions: []
        };
        
        // فحص الفوائد
        if (transaction.interest && transaction.interest > 0) {
            check.passed = false;
            check.issues.push({
                type: 'INTEREST_DETECTED',
                message: 'تم اكتشاف فائدة ربوية',
                value: transaction.interest
            });
            check.suggestions.push('إزالة الفائدة واستبدالها برسوم خدمة ثابتة');
        }
        
        // فحص ربا الفضل في المعادن الربوية
        if (this.isRibawiMetal(transaction.product)) {
            if (transaction.product.type === transaction.payment?.type) {
                if (transaction.product.weight !== transaction.payment?.weight) {
                    check.passed = false;
                    check.issues.push({
                        type: 'RIBA_FADL',
                        message: 'ربا الفضل: تبادل نفس المعدن بأوزان مختلفة'
                    });
                }
            }
        }
        
        return check;
    }
    
    /**
     * فحص الغرر
     */
    checkGharar(transaction) {
        const check = {
            rule: this.rules.gharar,
            passed: true,
            issues: [],
            suggestions: []
        };
        
        // فحص وصف المنتج
        if (!transaction.product?.description || 
            transaction.product.description.length < 50) {
            check.passed = false;
            check.severity = 'HIGH';
            check.issues.push({
                type: 'INSUFFICIENT_DESCRIPTION',
                message: 'وصف المنتج غير كافٍ'
            });
            check.suggestions.push('إضافة وصف تفصيلي للمنتج');
        }
        
        // فحص وضوح السعر
        if (!transaction.price || transaction.price <= 0) {
            check.passed = false;
            check.severity = 'CRITICAL';
            check.issues.push({
                type: 'UNKNOWN_PRICE',
                message: 'السعر غير محدد'
            });
        }
        
        // فحص القدرة على التسليم
        if (!transaction.deliveryDate) {
            check.issues.push({
                type: 'NO_DELIVERY_DATE',
                message: 'موعد التسليم غير محدد'
            });
            check.suggestions.push('تحديد موعد تسليم واضح');
        }
        
        return check;
    }
    
    /**
     * فحص الغش
     */
    checkGhish(transaction) {
        const check = {
            rule: this.rules.ghish,
            passed: true,
            issues: [],
            suggestions: []
        };
        
        // فحص دقة الوزن
        if (transaction.product?.weight) {
            const tolerance = 0.01; // 1% tolerance
            if (Math.abs(transaction.declaredWeight - transaction.actualWeight) / 
                transaction.declaredWeight > tolerance) {
                check.passed = false;
                check.issues.push({
                    type: 'WEIGHT_MISMATCH',
                    message: 'فرق كبير بين الوزن المُعلن والفعلي'
                });
            }
        }
        
        // فحص ذكر العيوب
        if (transaction.product?.hasDefects && !transaction.product?.defectsDisclosed) {
            check.passed = false;
            check.issues.push({
                type: 'HIDDEN_DEFECTS',
                message: 'عيوب غير مُفصح عنها'
            });
            check.suggestions.push('يجب الإفصاح عن جميع العيوب');
        }
        
        return check;
    }
    
    /**
     * فحص الاحتكار
     */
    checkIhtikar(transaction) {
        const check = {
            rule: this.rules.ihtikar,
            passed: true,
            issues: [],
            suggestions: []
        };
        
        // فحص كمية الشراء
        if (transaction.quantity > transaction.product?.maxPurchaseLimit) {
            check.passed = false;
            check.issues.push({
                type: 'EXCEEDS_LIMIT',
                message: 'الكمية تتجاوز الحد المسموح'
            });
            check.suggestions.push('تقليل الكمية لمنع الاحتكار');
        }
        
        return check;
    }
    
    /**
     * التحقق إذا كان المعدن من المعادن الربوية
     */
    isRibawiMetal(product) {
        const ribawiMetals = ['gold', 'silver', 'ذهب', 'فضة'];
        return ribawiMetals.includes(product?.type?.toLowerCase());
    }
    
    /**
     * حساب النتيجة
     */
    calculateScore(checks) {
        let score = 100;
        for (const check of checks) {
            if (!check.passed) {
                if (check.rule.severity === 'CRITICAL') {
                    score -= 50;
                } else if (check.rule.severity === 'HIGH') {
                    score -= 25;
                } else {
                    score -= 10;
                }
            }
        }
        return Math.max(0, score);
    }
    
    /**
     * إنهاء النتيجة وتسجيلها
     */
    finalizeResult(result) {
        // إضافة الفتوى
        result.fatwa = this.generateFatwa(result);
        
        // تسجيل في سجل التدقيق
        this.auditLog.push({
            ...result,
            loggedAt: new Date().toISOString()
        });
        
        return result;
    }
    
    /**
     * توليد الفتوى
     */
    generateFatwa(result) {
        if (result.status === 'APPROVED') {
            return {
                ruling: 'جائز',
                confidence: 'عالية',
                note: 'المعاملة متوافقة مع الشريعة الإسلامية',
                evidence: 'وَأَحَلَّ اللَّهُ الْبَيْعَ'
            };
        } else if (result.status === 'NEEDS_REVIEW') {
            return {
                ruling: 'يحتاج مراجعة',
                confidence: 'متوسطة',
                note: 'المعاملة تحتاج مراجعة من الهيئة الشرعية',
                issues: result.checks.filter(c => !c.passed).map(c => c.rule.name)
            };
        } else {
            return {
                ruling: 'غير جائز',
                confidence: 'عالية',
                note: 'المعاملة تتعارض مع أحكام الشريعة',
                violations: result.checks.filter(c => !c.passed).map(c => ({
                    rule: c.rule.name,
                    evidence: c.rule.evidence
                }))
            };
        }
    }
}

module.exports = { ShariaComplianceEngine, SHARIA_RULES };
```
