## ⑧ استخدام الوحدات الجديدة

### توليد هوية رقمية

```javascript
const { generateDID, generateKeyPair, signEntity, buildDIDDocument } = require('./lib/sheikha-id');

const did   = generateDID('user');      // did:sheikha:user:usr-a1b2c3d4
const keys  = generateKeyPair();
const doc   = buildDIDDocument(did, keys.publicKey, { nameAr: 'سلمان', maqsad: 'MAL' });
const signed = signEntity({ did, name: 'سلمان' }, keys.privateKey);
```

### التحقق الشرعي

```javascript
const { verifyCompliance, buildShariaRef } = require('./core/islamic-db');

const result = verifyCompliance({ hasRiba: false, hasGharar: false, hasMutualConsent: true });
// { compliant: true, violations: [], references: [] }

const ref = buildShariaRef('MAL');
// { maqsad: 'حفظ المال', verse: {...}, hadith: {...} }
```

### توجيه الطلبات عبر الموجّه العصبي

```javascript
const { router } = require('./core/engines');

const response = await router.route({
    intent: 'market',
    entity: { id: 'did:sheikha:user:usr-abc123' },
    data:   { commodity: 'ذهب', quantity: 100, unit: 'جرام' },
});
// response.schema === 'sheikha/v2'
// response.meta.sharia_ref  ← المرجع الشرعي
// response.meta.traceId     ← معرف التتبع
```
